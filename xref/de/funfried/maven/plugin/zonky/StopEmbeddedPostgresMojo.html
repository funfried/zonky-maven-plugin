20  /**
21   * Goal which stops the embedded postgres database.
22   */
23  @Mojo(name = "stop", defaultPhase = LifecyclePhase.NONE)
24  public class StopEmbeddedPostgresMojo extends AbstractMojo {
25  	/**
26  	 * The maven project.
27  	 */
28  	@Parameter(defaultValue = "${project}", required = true, readonly = true)
29  	private MavenProject project;
30  
31  	/**
32  	 * Contains the full list of projects in the reactor.
33  	 */
34  	@Parameter(defaultValue = "${reactorProjects}", readonly = true, required = true)
35  	private List<MavenProject> reactorProjects;
36  
37  	/**
38  	 * Stops the embedded postgres database.
39  	 *
40  	 * @throws MojoExecutionException if an error occurs
41  	 */
42  	@Override
43  	public void execute() throws MojoExecutionException {
44  		EmbeddedPostgres pg = MavenProjectUtil.getProjectProperty(project, reactorProjects, "zonky");
45  		if (pg != null) {
46  			String workDir = MavenProjectUtil.getProjectProperty(project, reactorProjects, "zonky.work.directory");
47  			String dataDir = MavenProjectUtil.getProjectProperty(project, reactorProjects, "zonky.data.directory");
48  
49  			File workDirFile = new File(workDir);
50  			File dataDirFile = new File(dataDir);
51  
52  			try {
53  				ZonkyUtil.stop(pg, workDirFile, dataDirFile);
54  
55  				stopped();
56  			} catch (IOException | InterruptedException | TimeoutException ex) {
57  				getLog().error("Failed to stop database", ex);
58  			}
59  		}
60  	}
61  
62  	private void stopped() {
63  		getLog().info("Stopped embedded postgres database at port " + project.getProperties().get("zonky.port") + " (JDBC URL: " + project.getProperties().get("zonky.jdbcUrl") + ")");
64  
65  		MavenProjectUtil.removeProjectProperty(project, reactorProjects,
66  				Set.of("zonky.host", "zonky.port", "zonky.database", "zonky.username", "zonky.password", "zonky.jdbcUrl", "zonky.work.directory", "zonky.data.directory", "zonky"));
67  	}
68  }
