19  /**
20   * Goal which stops the embedded postgres database.
21   */
22  @Mojo(name = "stop", defaultPhase = LifecyclePhase.NONE)
23  public class StopEmbeddedPostgresMojo extends AbstractMojo {
24  	/**
25  	 * The maven project.
26  	 */
27  	@Parameter(defaultValue = "${project}", required = true, readonly = true)
28  	private MavenProject project;
29  
30  	/**
31  	 * Contains the full list of projects in the reactor.
32  	 */
33  	@Parameter(defaultValue = "${reactorProjects}", readonly = true, required = true)
34  	private List<MavenProject> reactorProjects;
35  
36  	/**
37  	 * Stops the embedded postgres database.
38  	 *
39  	 * @throws MojoExecutionException if an error occurs
40  	 */
41  	@Override
42  	public void execute() throws MojoExecutionException {
43  		EmbeddedPostgres pg = MavenProjectUtil.getProjectProperty(project, reactorProjects, MavenProjectUtil.PROP_DB_INSTANCE);
44  		if (pg != null) {
45  			String workDir = MavenProjectUtil.getProjectProperty(project, reactorProjects, MavenProjectUtil.PROP_WORK_DIRECTORY);
46  			String dataDir = MavenProjectUtil.getProjectProperty(project, reactorProjects, MavenProjectUtil.PROP_DATA_DIRECTORY);
47  
48  			File workDirFile = new File(workDir);
49  			File dataDirFile = new File(dataDir);
50  
51  			try {
52  				ZonkyUtil.stop(pg, workDirFile, dataDirFile);
53  
54  				stopped();
55  			} catch (IOException | InterruptedException | TimeoutException ex) {
56  				getLog().error("Failed to stop database", ex);
57  			}
58  		}
59  	}
60  
61  	private void stopped() {
62  		getLog().info("Stopped embedded postgres database at port " + MavenProjectUtil.getProjectProperty(project, reactorProjects, MavenProjectUtil.PROP_PORT) + " (JDBC URL: "
63  				+ MavenProjectUtil.getProjectProperty(project, reactorProjects, MavenProjectUtil.PROP_JDBC_URL) + ")");
64  
65  		MavenProjectUtil.removeAllProjectProperties(project, reactorProjects);
66  	}
67  }
