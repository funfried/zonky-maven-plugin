28  /**
29   * Goal which starts an embedded postgres database.
30   */
31  @Mojo(name = "start", defaultPhase = LifecyclePhase.INITIALIZE, requiresProject = true, threadSafe = true)
32  public class StartEmbeddedPostgresMojo extends AbstractMojo {
33  	/**
34  	 * The port on which the database will be accessible. A value less than or equal to 0 means auto detect a free port. The port is available through the property ${zonky.port}.
35  	 */
36  	@Parameter(defaultValue = "0", property = "port")
37  	private int port;
38  
39  	/**
40  	 * If {@code true}, a create database statement with the given database name will be executed on startup.
41  	 */
42  	@Parameter(defaultValue = "true", property = "createDatabase")
43  	private boolean createDatabase;
44  
45  	/**
46  	 * The database name to write your data. Should not be postgres!
47  	 */
48  	@Parameter(defaultValue = "data", property = "databaseName")
49  	private String databaseName;
50  
51  	/**
52  	 * Define what should be done when the database is already started and the start goal is called again. Choose between:
53  	 * <ul>
54  	 * <li>fail (lets the build fail)</li>
55  	 * <li>reinit (drops the database and if "createDatabase" is true recreates the database again)</li>
56  	 * <li>restart (stops the database and starts it again)</li>
57  	 * <li>ignore (just keeps the current database and does not start a new one)</li>
58  	 * </ul>
59  	 */
60  	@Parameter(defaultValue = "reinit", property = "onAlreadyStarted")
61  	private AlreadyStartedPolicy onAlreadyStarted;
62  
63  	/**
64  	 * The working directory for the embedded database.
65  	 */
66  	@Parameter(defaultValue = "${project.build.directory}/embedded-postgres/work", property = "workingDirectory")
67  	private String workingDirectory;
68  
69  	/**
70  	 * The data directory for the embedded database.
71  	 */
72  	@Parameter(defaultValue = "${project.build.directory}/embedded-postgres/data", property = "dataDirectory")
73  	private String dataDirectory;
74  
75  	/**
76  	 * The maven project.
77  	 */
78  	@Parameter(defaultValue = "${project}", required = true, readonly = true)
79  	private MavenProject project;
80  
81  	/**
82  	 * Contains the full list of projects in the reactor.
83  	 */
84  	@Parameter(defaultValue = "${reactorProjects}", readonly = true, required = true)
85  	private List<MavenProject> reactorProjects;
86  
87  	/**
88  	 * Starts the embedded postgres database.
89  	 *
90  	 * @throws MojoExecutionException if an error occurs
91  	 */
92  	@Override
93  	public void execute() throws MojoExecutionException {
94  		EmbeddedPostgres pg = MavenProjectUtil.getProjectProperty(project, reactorProjects, "zonky");
95  		if (pg != null) {
96  			if (AlreadyStartedPolicy.fail.equals(onAlreadyStarted)) {
97  				throw new MojoExecutionException("Embedded database already started.");
98  			}
99  
100 			if (AlreadyStartedPolicy.reinit.equals(onAlreadyStarted)) {
101 				DataSource dataSource = pg.getDatabase("postgres", "postgres");
102 				try (Connection connection = dataSource.getConnection()) {
103 					try (Statement stmt = connection.createStatement()) {
104 						stmt.execute("DROP DATABASE \"" + databaseName + "\";");
105 
106 						if (createDatabase) {
107 							stmt.execute("CREATE DATABASE \"" + databaseName + "\";");
108 						}
109 					}
110 				} catch (SQLException ex) {
111 					throw new MojoExecutionException("Failed to reset embedded database", ex);
112 				}
113 
114 				getLog().info("Embedded postgres database reinitialized");
115 			} else if (AlreadyStartedPolicy.restart.equals(onAlreadyStarted)) {
116 				try {
117 					String workDir = project.getProperties().getProperty("zonky.work.directory");
118 					String dataDir = project.getProperties().getProperty("zonky.data.directory");
119 
120 					File workDirFile = new File(workDir);
121 					File dataDirFile = new File(dataDir);
122 
123 					ZonkyUtil.stop(pg, workDirFile, dataDirFile);
124 
125 					start(port, workDirFile, dataDirFile);
126 				} catch (IOException | InterruptedException | TimeoutException ex) {
127 					getLog().error("Failed to stop database", ex);
128 				}
129 			}
130 		} else {
131 			String subDir = UUID.randomUUID().toString();
132 
133 			start(port, new File(workingDirectory, subDir), new File(dataDirectory, subDir));
134 		}
135 	}
136 
137 	private EmbeddedPostgres start(int port, File workingDirectory, File dataDirectory) throws MojoExecutionException {
138 		EmbeddedPostgres pg;
139 
140 		try {
141 			pg = ZonkyUtil.start(port, workingDirectory, dataDirectory);
142 
143 			if (createDatabase) {
144 				DataSource dataSource = pg.getDatabase("postgres", "postgres");
145 				try (Connection connection = dataSource.getConnection()) {
146 					try (Statement stmt = connection.createStatement()) {
147 						stmt.execute("DROP DATABASE IF EXISTS \"" + databaseName + "\";");
148 						stmt.execute("CREATE DATABASE \"" + databaseName + "\";");
149 					}
150 				} catch (SQLException ex) {
151 					throw new MojoExecutionException("Failed to create embedded database '" + databaseName + "'", ex);
152 				}
153 			}
154 
155 			started(pg, workingDirectory, dataDirectory);
156 		} catch (IOException ex) {
157 			throw new MojoExecutionException("Failed to start embedded database", ex);
158 		}
159 
160 		return pg;
161 	}
162 
163 	private void started(EmbeddedPostgres pg, File workingDirectory, File dataDirectory) {
164 		int pgPort = pg.getPort();
165 		String jdbcUrl = pg.getJdbcUrl("postgres", databaseName);
166 
167 		getLog().info("Started embedded postgres database at port " + pgPort + " (JDBC URL: " + jdbcUrl + ")");
168 
169 		Map<String, Object> properties = new HashMap<>();
170 		properties.put("zonky.host", "localhost");
171 		properties.put("zonky.port", pgPort);
172 		properties.put("zonky.database", databaseName);
173 		properties.put("zonky.username", "postgres");
174 		properties.put("zonky.password", "postgres");
175 		properties.put("zonky.jdbcUrl", jdbcUrl);
176 		properties.put("zonky.work.directory", workingDirectory.getAbsolutePath());
177 		properties.put("zonky.data.directory", dataDirectory.getAbsolutePath());
178 		properties.put("zonky", pg);
179 
180 		MavenProjectUtil.putProjectProperty(project, reactorProjects, properties);
181 	}
