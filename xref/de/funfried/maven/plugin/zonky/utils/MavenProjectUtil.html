22  
23  	public static final String PROP_PASSWORD = "zonky.password";
24  
25  	public static final String PROP_JDBC_URL = "zonky.jdbcUrl";
26  
27  	public static final String PROP_WORK_DIRECTORY = "zonky.work.directory";
28

33  	public static final Set<String> PROPS_ALL = Set.of(PROP_HOST, PROP_PORT, PROP_DATABASE, PROP_USERNAME, PROP_PASSWORD, PROP_JDBC_URL, PROP_WORK_DIRECTORY, PROP_DATA_DIRECTORY, PROP_DB_INSTANCE);
34  
35  	private MavenProjectUtil() {
36  	}
37  
38  	public static <E extends Object> E getProjectProperty(MavenProject project, List<MavenProject> reactorProjects, String key) {
39  		E value = getProjectProperty(project, key);
40  		if (value == null) {
41  			for (MavenProject p : reactorProjects) {
42  				E v = getProjectProperty(p, key);
43  				if (v != null) {
44  					value = v;
45  
46  					break;
47  				}
48  			}
49  		}
50  
51  		return value;
52  	}
53  
54  	private static <E extends Object> E getProjectProperty(MavenProject project, String key) {
55  		return (E) project.getProperties().get(key);
56  	}
57  
58  	public static void putProjectProperty(MavenProject project, List<MavenProject> reactorProjects, Map<String, Object> properties) {
59  		for (Map.Entry<String, Object> item : properties.entrySet()) {
60  			project.getProperties().put(item.getKey(), item.getValue());
61  		}
62  
63  		for (MavenProject p : reactorProjects) {
64  			for (Map.Entry<String, Object> item : properties.entrySet()) {
65  				p.getProperties().put(item.getKey(), item.getValue());
66  			}
67  		}
68  	}
69  
70  	public static void removeProjectProperty(MavenProject project, List<MavenProject> reactorProjects, String propertyKey) {
71  		project.getProperties().remove(propertyKey);
72  
73  		for (MavenProject p : reactorProjects) {
74  			p.getProperties().remove(propertyKey);
75  		}
76  	}
77  
78  	public static void removeProjectProperties(MavenProject project, List<MavenProject> reactorProjects, Set<String> propertyKeys) {
79  		for (String key : propertyKeys) {
80  			project.getProperties().remove(key);
81  		}
82  
83  		for (MavenProject p : reactorProjects) {
84  			for (String key : propertyKeys) {
85  				p.getProperties().remove(key);
86  			}
87  		}
88  	}
89  
90  	public static void removeAllProjectProperties(MavenProject project, List<MavenProject> reactorProjects) {
91  		removeProjectProperties(project, reactorProjects, PROPS_ALL);
92  	}
93  }
